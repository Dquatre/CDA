);

   idSerie INT,
   serieName VARCHAR(50)  NOT NULL
);

CREATE TABLE games(
   idGame INT,
   title VARCHAR(50)  NOT NULL,
   subTitle VARCHAR(50) ,
   releaseYear INT,
   idSerie INT
);

    idGame_Platform INT,

   idPlatform INT
